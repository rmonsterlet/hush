import { Component, Optional, OnInit } from '@angular/core';
import { MdDialog, MdDialogRef, MdSnackBar } from '@angular/material';
import { HomeService } from './home.service';
import { UserComponent } from './user';
 
@Component({
  selector: 'app-home',
  templateUrl: './home.component.html',
  styleUrls: ['./home.component.css'],
  providers: [
    HomeService
  ]
})
export class HomeComponent implements OnInit {
 
  progress: number = 0;
  message: string;
 
  tiles = [
    { text: 'Zone 1', cols: 3, rows: 1, color: 'lightblue' },
    { text: 'Zone 2', cols: 1, rows: 2, color: 'lightgreen' },
    { text: 'Zone 3', cols: 1, rows: 1, color: 'lightpink' },
    { text: 'Zone 4', cols: 2, rows: 1, color: '#DDBDF1' },
  ]
 
  apps: any[] = [
    { name: 'Application 1', version: 'v0.1.2' },
    { name: 'Application 2', version: 'v4.2.2' },
    { name: 'Application 3', version: 'v1.3.0' },
  ]
 
  constructor(private homeService: HomeService, private dialog: MdDialog, private snackbar: MdSnackBar) {
    // Update the value for the progress-bar on an interval.
    setInterval(() => {
      this.progress = (this.progress + Math.floor(Math.random() * 4) + 1) % 100;
    }, 200);
  }
 
  ngOnInit() {
  }
 
  getPublicMsg() {
    this.homeService.getPublicMsg().then(result => this.message = result.str);
  }
 
  getPrivateMsg() {
    this.homeService.getPrivateMsg().then(result => this.message = result.str);
  }
 
  getTechnicalException() {
    this.homeService.getTechnicalException().catch(this.callbackFailure);
  }
 
  getFunctionalException() {
    this.homeService.getFunctionalException().catch(this.callbackFailure);
  }
 
  getOtherException() {
    this.homeService.getOtherException().catch(this.callbackFailure);
  }
 
  private callbackFailure(error) {
    var json = error.json();
    alert(json.message + ' Code : (' + json.codeErreur + ')');
  }
 
 
  lastDialogResult: string;
 
  openDialog() {
    let dialogRef = this.dialog.open(DialogContent);
 
    dialogRef.afterClosed().subscribe(result => {
      this.lastDialogResult = result;
    })
  }
 
  showSnackbar() {
    this.snackbar.open('Bienvenue à la Bnf ' + (this.lastDialogResult || ''), 'Fermer');
  }
 
  warn() {
    location.reload();
  }
}
