import { Injectable } from '@angular/core';
import { HttpDefaultService } from '../../_utils';
import { User } from './user';
 
import 'rxjs/add/operator/toPromise';
 
@Injectable()
export class UserService {
 
  constructor(private httpService: HttpDefaultService) { }
 
  addPersonne(user: User): Promise<any> {
 
    var split: any[] = user.birthDate.split('/');
    var dateNaissance = new Date(split[2], (split[1] - 1), split[0]);
 
    return this.httpService.post('rest/personnes/add', {
      nom: user.lastName,
      prenom: user.firstName,
      dateNaissance: dateNaissance
    });
  }
 
  getPersonnes() {
    return this.httpService.get('rest/personnes');
  }
}
